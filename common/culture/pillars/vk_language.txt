language_arberian = {
	type = language
	is_shown = {
		language_is_shown_trigger = {
			LANGUAGE = language_arberian
		}
	}
	ai_will_do = {
		value = 10
		if = {
			limit = { has_cultural_pillar = language_arberian }
			multiply = 10
		}
	}
	
	color = arberian
}
